import numpy as np from sklearn.linear_model import LinearRegression from sklearn.metrics import mean_squared_error, r2_score # Step 1: Create sample data house_sizes = np.array([1000, 1500, 2000, 2500, 3000]).reshape(-1, 1) prices = np.array([200000, 300000, 400000, 500000, 600000]) # Step 2: Train model model = LinearRegression() model.fit(house_sizes, prices) # Step 3: Make predictions predicted = model.predict(house_sizes) # Step 4: Evaluate r2 = r2_score(prices, predicted) mse = mean_squared_error(prices, predicted) print(f"R² Score: {r2:.3f}, MSE: {mse:.0f}")
