# Your First ML Model - Linear Regression import numpy as np import matplotlib.pyplot as plt from sklearn.linear_model import LinearRegression from sklearn.metrics import mean_squared_error, r2_score # Step 1: Create sample data house_sizes = np.array([1000, 1500, 2000, 2500, 3000]).reshape(-1, 1) prices = np.array([200000, 300000, 400000, 500000, 600000]) # Step 2: Create and train the model model = LinearRegression() model.fit(house_sizes, prices) # Step 3: Make predictions predicted_prices = model.predict(house_sizes) # Step 4: Evaluate performance mse = mean_squared_error(prices, predicted_prices) r2 = r2_score(prices, predicted_prices) print(f"R² Score: {r2:.3f}") print(f"MSE: {mse:.0f}") print(f"Model equation: Price = {model.coef_[0]:.0f} * Size + {model.intercept_:.0f}")
